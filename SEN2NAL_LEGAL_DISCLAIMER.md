```python
import streamlit as st

def show_footer_disclaimer():
    """Show footer disclaimer on every page."""
    st.markdown("""
    ---
    <div style="text-align: center; color: #888; font-size: 0.8em; padding: 20px;">
    ⚠️ <strong>Disclaimer:</strong> Sen2Nal is for informational purposes only 
    and is NOT financial advice. Investing involves risk of loss. 
    Past performance does not guarantee future results. 
    The creator(s) are not liable for any financial losses.
    <a href="/Disclaimer" target="_self">Read full disclaimer</a>
    </div>
    """, unsafe_allow_html=True)

def show_signal_warning():
    """Show warning when displaying trading signals."""
    st.warning("""
    ⚠️ **IMPORTANT:** This signal is for INFORMATIONAL PURPOSES ONLY.
    
    • NOT financial advice or a recommendation to buy/sell
    • Based on experimental sentiment analysis
    • Past performance does NOT guarantee future results
    • Always consult a licensed financial advisor
    • Never invest money you cannot afford to lose
    """)

def show_first_time_agreement():
    """Show agreement modal for first-time users."""
    if 'disclaimer_accepted' not in st.session_state:
        st.session_state.disclaimer_accepted = False
    
    if not st.session_state.disclaimer_accepted:
        st.markdown("## ⚠️ Important Notice")
        st.markdown("""
        Before using Sen2Nal, please read and acknowledge the following:
        
        1. **Sen2Nal is NOT a financial advisor** and does not provide investment advice.
        2. **You may lose money** by investing in securities.
        3. **Past performance does NOT guarantee future results.**
        4. **All investment decisions are YOUR responsibility.**
        5. The creator(s) bear **NO LIABILITY** for any financial losses.
        """)
        
        agree = st.checkbox("I understand and accept all terms")
        
        if agree:
            if st.button("Continue to Sen2Nal"):
                st.session_state.disclaimer_accepted = True
                st.rerun()
        else:
            st.stop()
```
